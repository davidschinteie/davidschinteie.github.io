                                cssDest     = 'css/',

                        gulp.task('browserSync', function() {
                            browserSync.init({
                                server: true
                            })
                        })

                        gulp.task('styles', function(){
                        return gulp.src(sassFiles)
                            .pipe(sass().on('error', sass.logError))
                            // alternative for minify css:
                            //.pipe(sass({outputStyle: 'compressed'}).on('error', sass.logError))
                            //.pipe(rename({ suffix: '.min' }))
                            .pipe(gulp.dest(cssDest))
                            //for browser sync
                            .pipe(browserSync.reload({
                                stream: true
                            }));
                        });

                            gulp.watch(sassFiles,['styles']);
                            //gulp.watch(testFiles, ['deploy']); - for ftp deploy

                            gulp.watch('**/*.html',browserSync.reload);
                            gulp.watch('js/**/*.js',browserSync.reload);
                        });

CSS Grid Layout

                        
                        -----Grid Columns-----
                        grid-template-columns: 300px 20% 30%;// 3 columns with 300px, 20% and 30% wide;
                        grid-template-columns: 30% 20% 50%;// 3 columns with 30%, 20% and 50% wide;
                        grid-template-columns: 1fr 1fr 1fr;// 3 columns with 1fr wide;
                        grid-template-columns: repeat(3, 1fr);// 3 columns with 1fr wide;
                        grid-template-columns: repeat(auto-fill, minmax(150px, 1fr));
                        grid-template-columns: repeat(auto-fit, minmax(150px, 1fr)); //Lesson 13 - responsive layouts
                        grid-template-columns: fit-content(100px) 150px 150px 150px; // fit-content(100px) == width auto but max-width: 130px;
                        grid-auto-columns: 100px;//the size of extra columns created
                        
                        -----Grid Rows-----
                        grid-auto-rows: minmax(150px, auto);//min row height 150px and max row height auto;
                        grid-template-rows: repeat(3, minmax(150px, auto));// 3 rows with grid-auto-rows: minmax(150px, auto);
                        grid-auto-rows: 100px;//the size of extra rows added;
                        grid-row-gap: 10px;//10px gap between rows
                        grid-column-gap: 10px;//10px gap between columns
                        grid-gap: 10px;//grid-row-gap: 10px and grid-column-gap: 10px;
                        
                        -----Grid Area-----
                        grid-template-areas:
                            "content    content     content"
                            "sidebar-1  sidebar-1   sidebar-2"
                            "footer     footer      footer";

                        -----Align & Justifying Items-----
                        align-items: start/center/end;//by default stretch value (aligning items up and down);
                        justify-items: start/center/end; //justify items left and right;
                        justify-self: start/center/end; //justify only one item;
                        align-self: start/center/end;// align only one item;

                        -----Grid Item/Element-----
                        grid-column-start:1;//Grid item starts at column line 1
                        grid-column-end: 3;//Grid item ends at column line 3
                        grid-column: 1/ 3;//grid-column-start: 1 and grid-column-end: 3;
                        grid-column: span 3;//use by default grid start line and expand it to the 3rd line;
                        grid-column: 1 / -1;//width 100% of the grid element
                        grid-row: 4/7;//grid-row-start: 1 (starts at row line 1) and grid-row-end: 3(ends at row line 3);
                        grid-row: 1 / -1;


                        grid-auto-flow: column;//it'll add the new items as columns not as rows as it is by default;
                        grid-auto-columns: 200px;//the size of auto items added;
