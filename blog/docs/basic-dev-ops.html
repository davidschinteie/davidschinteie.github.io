                    #Remove .html from the url's
                    index index.html;

                    if (!-f "${request_filename}index.html") {
                        rewrite ^/(.*)/$ /$1 permanent;
                    }

                    if ($request_uri ~* "/index.html") {
                        rewrite (?i)^(.*)index\.html$ $1 permanent;
                    }   

                    if ($request_uri ~* ".html") {
                        rewrite (?i)^(.*)/(.*)\.html $1/$2 permanent;
                    }

                    location / {
                    autoindex on;
                    try_files $uri.html $uri $uri/ /index.html;
